Module OSA::QuickTimePlayer::TXOP

Constants

SCROLL_IN = 'tfsi'   Scroll text in until all text is visible.
SCROLL_OUT = 'tfso'   Scroll text out until all text is gone.
SCROLL_HORIZONTAL = 'tfsh'   Scroll text horizontally (default is vertically).
SCROLL_REVERSE = 'tfsr'   Scroll text down or backwards (depending on current scroll direction).
SCROLL_CONTINUOUS = 'tfsc'   Scroll text out in response to next text scrolling in.
FLOW_HORIZONTAL = 'tffh'   Flow text in textbox.

[Validate]
